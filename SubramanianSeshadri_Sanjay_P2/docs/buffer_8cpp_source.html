71  void BufMgr::allocBuf(FrameId &frame)
72  {
73  std::uint32_t numPinnedPages = 0;
74  for (;;)
75  {
76  // Advancing clock pointer to the next frame
77  advanceClock();
78 
79  // Checking if the page is valid
80  if (bufDescTable[clockHand].valid)
81  {
82  // Page is valid
83  // Checking if refbit is set
84  if (!bufDescTable[clockHand].refbit)
85  {
86  // Refbit is not set
87  // Checking if pincount is 0 (i.e. the page
88  // is not used by any user)
89  if (bufDescTable[clockHand].pinCnt == 0)
90  {
91  // Page is not pinned, i.e. it is not
92  // used by any user
93 
94  // Checking if dirty bit is set
95  if (bufDescTable[clockHand].dirty)
96  {
97  // Page is dirty. Flush page to disk
98  bufDescTable[clockHand].file->writePage(bufPool[clockHand]);
99  hashTable->remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
100  }
101 
102  // Clearing the frame (i.e. pinCnt = 0, dirty = false;
103  // valid = false and refbit = false
104  bufDescTable[clockHand].Clear();
105  // Use the frame. Returning frame by reference
106  frame = bufDescTable[clockHand].frameNo;
107  return;
108  }
109  else if (bufDescTable[clockHand].pinCnt > 0)
110  {
111  // Page is pinned, i.e it is used,then
112  // it can't be used. Skip to next frame
113  numPinnedPages++;
114  if (numPinnedPages == numBufs)
115  {
116  throw BufferExceededException();
117  }
118  else
119  {
120  continue;
121  }
122  }
123  }
124  else
125  {
126  // Setting the refbit to false if set
127  bufDescTable[clockHand].refbit = false;
128  continue;
129  }
130  }
131  else
132  {
133  // Page is not valid
134  // Setting up the frame (i.e. pinCnt = 1, dirty = false;
135  // valid = true and refbit = true
136  bufDescTable[clockHand].Set(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
137 
138  // Use the frame. Returning frame by reference
139  frame = bufDescTable[clockHand].frameNo;
140  return;
141  }
